include irvine32.inc
.data
	row dword 4
	col dword 1 
	alp dword 65
	
	spacecnt dword 3
	spc byte " ",0
.code 
main proc
	mov ecx,row
	outer:
		mov ebx,ecx
		mov ecx, spacecnt
		space:
		mov edx, offset spc
		call writestring
		loop space
		mov al,alp
		mov ecx ,col
		inner:
		mov al,alp
		call writechar
		inc alp
		loop inner 
		add col,2
		mov ecx,ebx
		call crlf
		loop outer
	exit
	main endp
	end main
